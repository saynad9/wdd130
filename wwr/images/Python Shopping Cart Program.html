Certainly! Here's a Python program that implements a shopping cart system, allowing users to add items, remove them, and view the total price:

python
class ShoppingCart:
    def __init__(self):
        self.items = {}

    def add_item(self, name, price, quantity):
        if name in self.items:
            self.items[name]['quantity'] += quantity
        else:
            self.items[name] = {'price': price, 'quantity': quantity}
        print(f"Added {quantity} {name}(s) to the cart.")

    def remove_item(self, name, quantity):
        if name in self.items:
            if self.items[name]['quantity'] <= quantity:
                del self.items[name]
                print(f"Removed all {name}(s) from the cart.")
            else:
                self.items[name]['quantity'] -= quantity
                print(f"Removed {quantity} {name}(s) from the cart.")
        else:
            print(f"{name} not found in the cart.")

    def calculate_total(self):
        total = sum(item['price'] * item['quantity'] for item in self.items.values())
        return total

    def display_cart(self):
        if not self.items:
            print("Your shopping cart is empty.")
        else:
            print("\nShopping Cart:")
            for name, details in self.items.items():
                print(f"{name}: {details['quantity']} @ ${details['price']} each")
            print(f"Total Price: ${self.calculate_total():.2f}")

def main():
    cart = ShoppingCart()

    while True:
        print("\nMenu:")
        print("1. Add item")
        print("2. Remove item")
        print("3. View cart")
        print("4. Checkout")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == '1':
            name = input("Enter item name: ")
            price = float(input("Enter item price: $"))
            quantity = int(input("Enter item quantity: "))
            cart.add_item(name, price, quantity)
        elif choice == '2':
            name = input("Enter item name to remove: ")
            quantity = int(input("Enter quantity to remove: "))
            cart.remove_item(name, quantity)
        elif choice == '3':
            cart.display_cart()
        elif choice == '4':
            print(f"\nCheckout Total: ${cart.calculate_total():.2f}")
            break
        elif choice == '5':
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()

How It Works:

  • ShoppingCart Class: Manages the items in the cart using a dictionary where the key is the item name, and the value is another dictionary containing the price and quantity.

  • add_item(name, price, quantity): Adds the specified quantity of the item to the cart. If the item already exists, it updates the quantity.

  • remove_item(name, quantity): Removes the specified quantity of the item from the cart. If the quantity to remove exceeds or equals the current quantity, the item is completely removed.

  • calculate_total(): Computes the total price of all items in the cart.

  • display_cart(): Displays all items in the cart along with their quantities and prices, and shows the total price.
